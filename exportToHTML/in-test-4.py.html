# неубывание 1+2+3 / 3+4+5
def calc1(check):
    results = {}
    for n in range(10000,99999):
        s = str(n)
        n1 = int(s[0]) + int(s[1]) + int(s[2])
        n2 = int(s[2]) + int(s[3]) + int(s[4])
        if (n1 > n2):
            res = str(n2) + str(n1)
        else:
            res = str(n1) + str(n2)
        if res in check:
            results[res] = res
            print(s, res)
    return results
